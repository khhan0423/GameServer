137 XLSheetState visibility() const
138 {
139 XLQuery query(XLQueryType::QuerySheetVisibility);
140 query.setParam("sheetID", relationshipID());
141 auto state = parentDoc().execQuery(query).template result<std::string>();
142 auto result = XLSheetState::Visible;
143
144 if (state == "visible" || state.empty()) {
145 result = XLSheetState::Visible;
146 }
147 else if (state == "hidden") {
148 result = XLSheetState::Hidden;
149 }
150 else if (state == "veryHidden") {
151 result = XLSheetState::VeryHidden;
152 }
153
154 return result;
155 }
156
161 void setVisibility(XLSheetState state)
162 {
163 auto stateString = std::string();
164 switch (state) {
165 case XLSheetState::Visible:
166 stateString = "visible";
167 break;
168
169 case XLSheetState::Hidden:
170 stateString = "hidden";
171 break;
172
173 case XLSheetState::VeryHidden:
174 stateString = "veryHidden";
175 break;
176 }
177
178 parentDoc().execCommand(XLCommand(XLCommandType::SetSheetVisibility)
179 .setParam("sheetID", relationshipID())
180 .setParam("sheetVisibility", stateString));
181 }
