112 std::is_same_v<std::decay_t<T>, std::string_view> || std::is_same_v<std::decay_t<T>, const char*> ||
113 std::is_same_v<std::decay_t<T>, char*> || std::is_same_v<T, XLDateTime>>::type* = nullptr>
114 XLCellValue(T value) // NOLINT
115 {
116 // ===== If the argument is a bool, set the m_type attribute to Boolean.
117 if constexpr (std::is_integral_v<T> && std::is_same_v<T, bool>) {
118 m_type = XLValueType::Boolean;
119 m_value = value;
120 }
121
122 // ===== If the argument is an integral type, set the m_type attribute to Integer.
123 else if constexpr (std::is_integral_v<T> && !std::is_same_v<T, bool>) {
124 m_type = XLValueType::Integer;
125 m_value = int64_t(value);
126 }
127
128 // ===== If the argument is a string type (i.e. is constructable from *char),
129 // ===== set the m_type attribute to String.
130 else if constexpr (std::is_same_v<std::decay_t<T>, std::string> || std::is_same_v<std::decay_t<T>, std::string_view> ||
131 std::is_same_v<std::decay_t<T>, const char*> ||
132 (std::is_same_v<std::decay_t<T>, char*> && !std::is_same_v<T, bool>))
133 {
134 m_type = XLValueType::String;
135 m_value = std::string(value);
136 }
137
138 // ===== If the argument is an XLDateTime, set the value to the date/time serial number.
139 else if constexpr (std::is_same_v<T, XLDateTime>) {
140 m_type = XLValueType::Float;
141 m_value = value.serial();
142 }
143
144 // ===== If the argument is a floating point type, set the m_type attribute to Float.
145 // ===== If not, a static_assert will result in compilation error.
146 else {
147 static_assert(std::is_floating_point_v<T>, "Invalid argument for constructing XLCellValue object");
148 if (std::isfinite(value)) {
149 m_type = XLValueType::Float;
150 m_value = double(value);
151 }
152 else {
153 m_type = XLValueType::Error;
154 m_value = std::string("#NUM!");
155 }
156 }
157 }

243
244 if constexpr (std::is_floating_point_v<T>) {
245 if (m_type == XLValueType::Error) return std::nan("1");

252 return std::get<std::string>(m_value).c_str();
253
254 if constexpr (std::is_same_v<T, XLDateTime>) return XLDateTime(std::get<double>(m_value));

341 template<
342 typename T,
343 typename std::enable_if<std::is_integral_v<T> || std::is_floating_point_v<T> || std::is_same_v<std::decay_t<T>, std::string> ||
344 std::is_same_v<std::decay_t<T>, std::string_view> || std::is_same_v<std::decay_t<T>, const char*> ||
345 std::is_same_v<std::decay_t<T>, char*> || std::is_same_v<T, XLCellValue> ||
346 std::is_same_v<T, XLDateTime>>::type* = nullptr>
347 XLCellValueProxy& operator=(T value)
348 { // NOLINT
349
350 if constexpr (std::is_integral_v<T> && std::is_same_v<T, bool>) // if bool
351 setBoolean(value);
352
353 else if constexpr (std::is_integral_v<T> && !std::is_same_v<T, bool>) // if integer
354 setInteger(value);
355
356 else if constexpr (std::is_floating_point_v<T>) // if floating point
357 setFloat(value);
358
359 else if constexpr (std::is_same_v<T, XLDateTime>)
360 setFloat(value.serial());
361
362 else if constexpr (std::is_same_v<std::decay_t<T>, std::string> || std::is_same_v<std::decay_t<T>, std::string_view> ||
363 std::is_same_v<std::decay_t<T>, const char*> ||
364 (std::is_same_v<std::decay_t<T>, char*> && !std::is_same_v<T, bool> && !std::is_same_v<T, XLCellValue>))
365 {
366 if constexpr (std::is_same_v<std::decay_t<T>, const char*> || std::is_same_v<std::decay_t<T>, char*>)
367 setString(value);
368 else if constexpr (std::is_same_v<std::decay_t<T>, std::string_view>)
369 setString(std::string(value).c_str());
370 else
371 setString(value.c_str());
372 }
373
374 if constexpr (std::is_same_v<T, XLCellValue>) {
375 switch (value.type()) {
376 case XLValueType::Boolean:
377 setBoolean(value.template get<bool>());
378 break;
379 case XLValueType::Integer:
380 setInteger(value.template get<int64_t>());
381 break;
382 case XLValueType::Float:
383 setFloat(value.template get<double>());
384 break;
385 case XLValueType::String:
386 setString(value.template get<const char*>());
387 break;
388 case XLValueType::Empty:
389 clear();
390 break;
391 default:
392 setError("#N/A");
393 break;
394 }
395 }
396
397 return *this;
398 }
