247 template<typename T,
248 typename std::enable_if<!std::is_same_v<T, XLRowDataProxy> &&
249 std::is_base_of_v<typename std::bidirectional_iterator_tag,
250 typename std::iterator_traits<typename T::iterator>::iterator_category>,
251 T>::type* = nullptr>
252 XLRowDataProxy& operator=(const T& values)
253 {
254 if (values.size() > MAX_COLS) throw XLOverflowError("Container size exceeds maximum number of columns.");
255 if (values.size() == 0) return *this;
256
257 // ===== If the container value_type is XLCellValue, the values can be copied directly.
258 if constexpr (std::is_same_v<typename T::value_type, XLCellValue>) {
259 // ===== First, delete the values in the first N columns.
260 deleteCellValues(values.size());
261
262 // ===== Then, prepend new cell nodes to current row node
263 auto colNo = values.size();
264 for (auto value = values.rbegin(); value != values.rend(); ++value) { // NOLINT
265 prependCellValue(*value, colNo);
266 --colNo;
267 }
268 }
269
270 // ===== If the container value_type is a POD type, use the overloaded operator= on each cell.
271 else {
272 auto range = XLRowDataRange(*m_rowNode, 1, values.size(), getSharedStrings());
273 auto dst = range.begin();
274 auto src = values.begin();
275
276 while (true) {
277 dst->value() = *src;
278 ++src;
279 if (src == values.end()) break;
280 ++dst;
281 }
282 }
283
284 return *this;
285 }

395 template<
396 typename Container,
397 typename std::enable_if<!std::is_same_v<Container, XLRowDataProxy> &&
398 std::is_base_of_v<typename std::bidirectional_iterator_tag,
399 typename std::iterator_traits<typename Container::iterator>::iterator_category>,
400 Container>::type* = nullptr>
401 Container convertContainer() const
402 {
403 Container c;
404 auto it = std::inserter(c, c.end());
405 for (const auto& v : getValues()) {
406 // ===== If the value_type of the container is XLCellValue, the value can be assigned directly.
407 if constexpr (std::is_same_v<typename Container::value_type, XLCellValue>) *it++ = v;
408
409 // ===== If the value_type is something else, the underlying value has to be extracted from the XLCellValue object.
410 // ===== Note that if the type contained in the XLCellValue object does not match the value_type, a bad_variant_access
411 // ===== exception will be thrown.
412 else
413 *it++ = v.get<typename Container::value_type>();
414 }
415 return c;
416 }
